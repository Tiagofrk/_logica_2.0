Algoritmo "media de 3 provas"

Var

   n1,n2,n3, media: real

Inicio
      n1<- n1 * 2
      escreva("Digite a primeira nota: ")
      leia(n1)
      n2<- n2 * 3
      escreva("Digite a segunda nota: ")
      leia(n2)
      n3<- n3 * 5
      escreva("Digite a terceira nota: ")
      leia(n3)

      media<- (n1 + n2+ n3) / 3
      escreval("A média final do aluno foi de", media:4:0)
      
      se (media >=7)entao
         escreva("O aluno foi aprovado !")
      senao (media <7)entao
            escreva("O aluno foi reprovado :(")
      fimse
